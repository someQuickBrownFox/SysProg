    bare = false
    logallrefupdates = true

    email="somequickbrownfox@googlemail.com"
[remote "origin"]
    fetch = +refs/heads/*:refs/remotes/origin/*
    url = git@github.com:someQuickBrownFox/SysProg.git
[branch "master"]
    remote = origin
    merge = refs/heads/master
